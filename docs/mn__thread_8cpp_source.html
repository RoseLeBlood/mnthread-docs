74 int basic_thread::create(int uiCore) {
75  if(!m_bMutexInit) {
76  m_runningMutex = new LockType_t();
77  m_contextMutext = new LockType_t();
78 
79  m_continuemutex = new LockType_t();
80  m_continuemutex2 = new LockType_t();
81 
82 
83  if(m_runningMutex->create() != ERR_MUTEX_OK)
84  return ERR_THREAD_CANTINITMUTEX;
85  if(m_contextMutext->create() != ERR_MUTEX_OK)
86  return ERR_THREAD_CANTINITMUTEX;
87 
88  if(m_continuemutex->create() != ERR_MUTEX_OK)
89  return ERR_THREAD_CANTINITMUTEX;
90  if(m_continuemutex2->create() != ERR_MUTEX_OK)
91  return ERR_THREAD_CANTINITMUTEX;
92 
93  }
94  m_continuemutex->lock();
95  m_runningMutex->lock();
96  if (m_bRunning)
97  {
98  m_runningMutex->unlock();
99  m_continuemutex->unlock();
100  return ERR_THREAD_ALREADYRUNNING;
101  }
102  m_runningMutex->unlock();
103 
104  if(uiCore == -1) {
105  xTaskCreate(&runtaskstub, m_strName,
106  m_usStackDepth,
107  this, m_uiPriority, &handle);
108  } else {
109  xTaskCreatePinnedToCore(&runtaskstub, m_strName,
110  m_usStackDepth,
111  this, m_uiPriority, &handle, uiCore);
112 
113  }
114  m_iCore = uiCore;
115  m_iID = get_new_id();
116 
117  if (handle == 0) {
118  m_continuemutex->unlock();
119  return ERR_THREAD_CANTSTARTTHREAD;
120  }
121 
122  m_continuemutex->unlock();
123 
124  on_create();
125 
126  return ERR_THREAD_OK;
127 }

129  m_continuemutex->lock();

131
