54 int basic_mutex::lock(unsigned int timeout) {
55  BaseType_t success;
56 
57  if (m_pmutex == NULL)
58  return ERR_MUTEX_NOTINIT;
59 
60  if (xPortInIsrContext()) {
61  BaseType_t xHigherPriorityTaskWoken = pdFALSE;
62  success = xSemaphoreTakeFromISR( m_pmutex, &xHigherPriorityTaskWoken );
63  if(xHigherPriorityTaskWoken)
64  _frxt_setup_switch();
65  } else {
66  success = xSemaphoreTake(m_pmutex, timeout);
67  }
68 
69  return success == pdTRUE ? ERR_MUTEX_OK : ERR_MUTEX_LOCK;
70 }
