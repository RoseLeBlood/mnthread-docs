52  void write(buffer_item_t item, TickType_t xTicksToWait) {
53  TickType_t xTicksEnd = xTaskGetTickCount() + xTicksToWait;
54  TickType_t xTicksRemaining = xTicksToWait;
55 
56  while (xTicksRemaining <= xTicksToWait) {
57  if(m_mLock->lock(xTicksRemaining) != NO_ERROR){
58  break;
59  }
60 
61  m_pBuffer[m_iHead] = item;
62 
63  if(m_bFull)
64  m_iTail = (m_iTail + 1) % m_szRingBufferSize;
65 
66  m_iHead = (m_iHead + 1) % m_szRingBufferSize;
67  m_bFull = (m_iHead == m_iTail);
68 
69  if (xTicksToWait != portMAX_DELAY) {
70  xTicksRemaining = xTicksEnd - xTaskGetTickCount();
71  }
72  m_mLock->unlock();
73  }
74  }
83  int write(buffer_item_t* items, unsigned int offset, unsigned int size, TickType_t xTicksToWait) {
84  if (size <= offset) return -1;

98  buffer_item_t read(TickType_t xTicksToWait) {
99  TickType_t xTicksEnd = xTaskGetTickCount() + xTicksToWait;
100  TickType_t xTicksRemaining = xTicksToWait;
101 
102  buffer_item_t _ret;
103  while (xTicksRemaining <= xTicksToWait) {
104  if(m_mLock->lock(xTicksRemaining) != NO_ERROR){
105  break;
106  }
107 
108  if(is_empty()) {
109  _ret = buffer_item_t();

115  m_bFull = false;
116  m_iTail = (m_iTail + 1) % m_szRingBufferSize;
117 
118  if (xTicksToWait != portMAX_DELAY) {
119  xTicksRemaining = xTicksEnd - xTaskGetTickCount();
120  }
121  m_mLock->unlock();
122  }
