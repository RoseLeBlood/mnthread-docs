50  void write(buffer_item_t item, TickType_t xTicksToWait) {
51  TickType_t xTicksEnd = xTaskGetTickCount() + xTicksToWait;
52  TickType_t xTicksRemaining = xTicksToWait;
53 
54  while (xTicksRemaining <= xTicksToWait) {
55  if(m_mLock->lock(xTicksRemaining) != NO_ERROR){
56  break;
57  }
58 
59  m_pBuffer[m_iHead] = item;
60 
61  if(m_bFull)
62  m_iTail = (m_iTail + 1) % m_szRingBufferSize;
63 
64  m_iHead = (m_iHead + 1) % m_szRingBufferSize;
65  m_bFull = (m_iHead == m_iTail);
66 
67  if (xTicksToWait != portMAX_DELAY) {
68  xTicksRemaining = xTicksEnd - xTaskGetTickCount();
69  }
70  m_mLock->unlock();
71  }
72  }
