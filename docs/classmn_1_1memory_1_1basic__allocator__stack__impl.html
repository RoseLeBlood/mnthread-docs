template<int TBUFFERSIZE>
class mn::memory::basic_allocator_stack_impl< TBUFFERSIZE >

Stack based allocator.

Note
- operates on buffer of TBUFFERSIZE bytes of stack memory
- never frees memory
- cannot be copied
