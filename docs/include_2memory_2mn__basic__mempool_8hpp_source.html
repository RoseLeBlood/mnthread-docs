166  virtual type_t* allocate(bool oFreedSelf = false, unsigned long xTicksToWait = portMAX_DELAY) {
167  if(m_vChunks.size() == 0) { return NULL; }

175  (it != m_vChunks.end() && (xTicksRemaining <= xTicksToWait) ); it++) {

194  xTicksRemaining = xTicksEnd - xTaskGetTickCount();
